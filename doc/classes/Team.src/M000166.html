  def initialize(abrev)
    if (abrev && abrev != '')
      @abrev = abrev
      if Team.teams[@abrev]
        @city = Team.teams[@abrev][0]
        @name = Team.teams[@abrev][1]
        if Team.teams[@abrev].length > 2
          @league = Team.teams[@abrev][2]
        end
      else
        @city = @abrev
        @name = @abrev
        @league = ''
      end
    else
        @city = ''
        @name = ''
        @league = ''
    end
  end
