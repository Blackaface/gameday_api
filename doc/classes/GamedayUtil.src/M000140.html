  def self.get_connection(url)
    self.read_config
    begin
      if !@@proxy_addr.empty?
        connection = open(url, :proxy => "http://#{@@proxy_addr}:#{@@proxy_port}")
      else
        connection = open(url)
      end
      connection
    rescue
      puts 'Could not open connection'
    end
  end
